Module Ast

type ident = string
type ty =
  1. | TInt
  2. | TBool
  3. | TReal
  4. | TCustom of string
type binop =
  1. | Add
  2. | Sub
  3. | Mul
  4. | Div
  5. | Eq
  6. | Neq
  7. | Lt
  8. | Le
  9. | Gt
  10. | Ge
  11. | And
  12. | Or
type unop =
  1. | Neg
  2. | Not
type loc = {
  1. line : int;
  2. col : int;
  3. line_end : int;
  4. col_end : int;
}
type iexpr = {
  1. iexpr : iexpr_desc;
  2. loc : loc option;
}
and iexpr_desc =
  1. | ILitInt of int
  2. | ILitBool of bool
  3. | IVar of ident
  4. | IBin of binop * iexpr * iexpr
  5. | IUn of unop * iexpr
  6. | IPar of iexpr
type hexpr =
  1. | HNow of iexpr
  2. | HPreK of iexpr * int
type relop =
  1. | REq
  2. | RNeq
  3. | RLt
  4. | RLe
  5. | RGt
  6. | RGe
type fo =
  1. | FTrue
  2. | FFalse
  3. | FRel of hexpr * relop * hexpr
  4. | FPred of ident * hexpr list
  5. | FNot of fo
  6. | FAnd of fo * fo
  7. | FOr of fo * fo
  8. | FImp of fo * fo
type 'a ltl =
  1. | LTrue
  2. | LFalse
  3. | LAtom of 'a
  4. | LNot of 'a ltl
  5. | LAnd of 'a ltl * 'a ltl
  6. | LOr of 'a ltl * 'a ltl
  7. | LImp of 'a ltl * 'a ltl
  8. | LX of 'a ltl
  9. | LG of 'a ltl
  10. | LW of 'a ltl * 'a ltl
type fo_ltl = fo ltl
type origin =
  1. | UserContract
  2. | Instrumentation
  3. | Coherency
  4. | Compatibility
  5. | AssumeAutomaton
  6. | Internal
type fo_o = {
  1. value : fo;
  2. origin : origin option;
  3. oid : int;
  4. loc : loc option;
}
type stmt = {
  1. stmt : stmt_desc;
  2. loc : loc option;
}
and stmt_desc =
  1. | SAssign of ident * iexpr
  2. | SIf of iexpr * stmt list * stmt list
  3. | SMatch of iexpr * (ident * stmt list) list * stmt list
  4. | SSkip
  5. | SCall of ident * iexpr list * ident list
type invariant_user = {
  1. inv_id : ident;
  2. inv_expr : hexpr;
}
val pp_invariant_user : Ppx_deriving_runtime.Format.formatter -> invariant_user -> Ppx_deriving_runtime.unit
val show_invariant_user : invariant_user -> Ppx_deriving_runtime.string
type invariant_state_rel = {
  1. is_eq : bool;
  2. state : ident;
  3. formula : fo;
}
val pp_invariant_state_rel : Ppx_deriving_runtime.Format.formatter -> invariant_state_rel -> Ppx_deriving_runtime.unit
val show_invariant_state_rel : invariant_state_rel -> Ppx_deriving_runtime.string
type atom_ltl = ident ltl
type vdecl = {
  1. vname : ident;
  2. vty : ty;
}
type node_attrs = {
  1. uid : int option;
  2. invariants_user : invariant_user list;
  3. invariants_state_rel : invariant_state_rel list;
  4. coherency_goals : fo_o list;
}
type transition_attrs = {
  1. uid : int option;
  2. ghost : stmt list;
  3. instrumentation : stmt list;
  4. warnings : string list;
}
type transition = {
  1. src : ident;
  2. dst : ident;
  3. guard : iexpr option;
  4. requires : fo_o list;
  5. ensures : fo_o list;
  6. body : stmt list;
  7. attrs : transition_attrs;
}
type node = {
  1. nname : ident;
  2. inputs : vdecl list;
  3. outputs : vdecl list;
  4. assumes : fo_ltl list;
  5. guarantees : fo_ltl list;
  6. instances : (ident * ident) list;
  7. locals : vdecl list;
  8. states : ident list;
  9. init_state : ident;
  10. trans : transition list;
  11. attrs : node_attrs;
}
type node_semantics = {
  1. sem_nname : ident;
  2. sem_inputs : vdecl list;
  3. sem_outputs : vdecl list;
  4. sem_instances : (ident * ident) list;
  5. sem_locals : vdecl list;
  6. sem_states : ident list;
  7. sem_init_state : ident;
  8. sem_trans : transition list;
}
type node_specification = {
  1. spec_assumes : fo_ltl list;
  2. spec_guarantees : fo_ltl list;
  3. spec_invariants_state_rel : invariant_state_rel list;
}
type program = node list
val semantics_of_node : node -> node_semantics
val specification_of_node : node -> node_specification
val show_program : program -> string
