Module Product_types

type product_state = {
  1. prog_state : Ast.ident;
  2. assume_state : int;
  3. guarantee_state : int;
}
type step_class =
  1. | Safe
  2. | Bad_assumption
  3. | Bad_guarantee
type automaton_edge = Automaton_engine.transition
type product_step = {
  1. src : product_state;
  2. dst : product_state;
  3. prog_transition : Abstract_model.transition;
  4. prog_guard : Ast.fo;
  5. assume_edge : automaton_edge;
  6. assume_guard : Ast.fo;
  7. guarantee_edge : automaton_edge;
  8. guarantee_guard : Ast.fo;
  9. step_class : step_class;
}
type prune_reason =
  1. | Incompatible_program_assumption
  2. | Incompatible_program_guarantee
  3. | Incompatible_assumption_guarantee
type pruned_step = {
  1. src : product_state;
  2. prog_transition : Abstract_model.transition;
  3. prog_guard : Ast.fo;
  4. assume_edge : automaton_edge;
  5. assume_src : int;
  6. assume_dst : int;
  7. assume_guard : Ast.fo;
  8. guarantee_edge : automaton_edge;
  9. guarantee_src : int;
  10. guarantee_dst : int;
  11. guarantee_guard : Ast.fo;
  12. reason : prune_reason;
}
type exploration = {
  1. initial_state : product_state;
  2. states : product_state list;
  3. steps : product_step list;
  4. pruned_steps : pruned_step list;
}
val compare_state : product_state -> product_state -> int
